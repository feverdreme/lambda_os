79/* mmap entry, type is stored in least significant tetrad (half byte) of size
80 * this means size described in 16 byte units (not a problem, most modern
81 * firmware report memory in pages, 4096 byte units anyway). */
82typedef struct {
83 uint64_t ptr;
84 uint64_t size;
85} _pack MMapEnt;
86#define MMapEnt_Ptr(a) ((a)->ptr)
87#define MMapEnt_Size(a) ((a)->size & 0xFFFFFFFFFFFFFFF0)
88#define MMapEnt_Type(a) ((a)->size & 0xF)
89#define MMapEnt_IsFree(a) (((a)->size&0xF)==1)
90
91#define MMAP_USED 0 /* don't use. Reserved or unknown regions */
92#define MMAP_FREE 1 /* usable memory */
93#define MMAP_ACPI 2 /* acpi memory, volatile and non-volatile as well */
94#define MMAP_MMIO 3 /* memory mapped IO region */

96#define INITRD_MAXSIZE 16 /* Mb */
